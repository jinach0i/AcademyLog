생성자 함수

생성자 함수를 이용하면 특정 속성과 메서드를 가진 객체를 쉽게 생성할 수 있으며 , 정의된 생성자 함수가 완성되면, 어디서든 new 연산자를 활용하여 그 속성과 메서드르 지닌 객체를 생성할 수 있음

    <script>
     //<![CDATA[
       function comFactory(cpuInfo, ramInfo, hddInfo){    //1.생성자 함수
           this.cpu=cpuInfo; //this는 생성될 객체
           this.ram=ramInfo;
           this.hdd=hddInfo;

           this.totInfo=function(){    //객체 메서드
               document.write("제조업체:한국전자 <br />");
               document.write("시스템종류:64비트 운영체제 <br />")
          }
	    }
       var speedCom=new comFactory("2.8GHZ","2.0GB","1TB");    //2.객체 생성
       document.write(speedCom.cpu+"<br />");
       document.write(speedCom.ram+"<br />");

      speedCom.totInfo();
     //]]>
    </script>

기존 방식(Dom Level 0) 이벤트 문제점

-위의 같이 객체에 이벤트를 직접 연결하여 작동하는 고전적인 방식을 'DOM Level 0'이라고 함. 객체 하나와 이벤트 하나를 연결하여 작동하는 이 방식은 한 객체에 동시에 같은 이벤트를 두 번 이상 적용할 수 없다는 문제를 안고 있으며  한 객체에 같은 이벤트를 두 번 이상 적용하게 되면 제일 마지막에 적용된 이벤트만 실행됨 

<script>
     //<![CDATA[
        function theFnc(){
            alert("javascript");
	    }
       window.onload=function(){    //작동 안됨
           var myBtn=document.getElementById("btn1");
           myBtn.onclick=theFnc;
       }

       window.onload=function(){    //같은 객체에 동일 이벤트 적용 작동 됨
           var myBtn=document.getElementById("btn2");
           myBtn.onclick=theFnc;
       }
     //]]>
    </script> 

    <body>
		<button id="btn1">버튼</button>
		<button id="btn2">버튼</button>
   </body>

DOM Level 2 방식 이벤트

한객체에 같은 이벤트를 두 번 적용할수있는 방식이지만 웹표준안과 비표준안(IE8 이하) 둘다 적용하기 위한 방법 

if(window.addEventListener){    //표준 방식 브라우저(크롭, 파이어폭스 등)일 경우
     표쥰 이벤트 등록 메서드;
}else if(window.attachEvent){    //비표준 방식 브라우저(IE8 이하)일 경우
    비표준 이벤트 등록 메서드;
}

표준방식 브라우저에대한 이벤트 등록방법

이벤트대상.addEventListener("이벤트명",함수명, false);

마지막 인자값false는 해당 이벤트를 끝내고 다음 이벤트로 넘기는 역할. 인자값 false가 포함되면 해당 이벤트가 발생한 위치에서 멈추고 아래로 진행되는데 이를 '버블업'방식이라고 함

비표준 방식 브라우저(IE8 이하)에 대한 이벤트 등록방법

이벤트대상.attachEvent("on"+"이벤트명", 함수명)

여기서 주의 할 점은 인자값에 이벤트명을 입력할 때 'on'을 붙여야 한다는 것이며, 버블방식의 설정(false)은 필요 없음 

    <script>
     //<![CDATA[
        function theFnc(){
            alert("javascript");
	    }
       function clickBtn1(){    //작동 안됨
           var myBtn1=document.getElementById("btn1");
           myBtn1.onclick=theFnc;
       }

       function clickBtn2(){    //같은 객체에 동일 이벤트 적용 작동 됨
           var myBtn2=document.getElementById("btn2");
           myBtn2.onclick=theFnc;
       }
       if(window.addEventListener){
            window.addEventListener("load",clickBtn1,false);
            window.addEventListener("load",clickBtn2,false);
       }else if(window.attachEvent){
            window.attachEvent("onload",clickBtn1);
            window.attachEvent("onload",clickBtn2);
     //]]>
    </script> 

    <body>
		<button id="btn1">버튼</button>
		<button id="btn2">버튼</button>
   </body>

이벤트 객체 속성

click, mousemove, mouseover ,mouseup, mousedown

clientX    현재 문서 기준, 이벤트가 발생한 X좌표(스크롤바 너비 계산 x)

clientY    현재 문서 기준, 이벤트가 발생한 Y좌표(스크롤바 높이 계산 x)

screenX   모니터 기준, 이벤트가 발생한 x좌표

pageX     현재 문서 기준, 이벤트가 발생한 X좌표(스크롤바 너비 계산 o)

pageY     현재 문서 기준, 이벤트가 발생한 Y좌표(스크롤바 높이 계산 o)

layerX     position 속성값이 'absolute'인 레이어 안에서의 상대 X좌표

layerY     position 속성값이 'absolute'인 레이어 안에서의 상대 Y좌표

target    이벤트가 발생한 대상 요소

button   마우스 버튼(왼쪽/오른쪽/휠)의 클릭된 상태 체크

    <script>
     //<![CDATA[
        document.onmousemove=function(e){
            var theEvent = e?e:window.event;    //이벤트 객체 생성
            var myForm=document.form1;
       
            myForm.clt_x.value="clientX:"+theEvent.clientX;

            myForm.pg_x.value="pageX:"+theEvent.pageX;
            myForm.pg_y.value="pageY:"+theEvent.pageY;
	    }
     //]]>
    </script>

theEvent.clientY와theEvent.pageY의 차이점

페이지에서 가로/세로 스크롤바가 발생했을 경우, client는 그 스크롤바 높이를 계산하지 않고, page는 스크롤바 높이 까지 계산하여 각각 위치값을 구함

theEvent.screenX    -모니터 좌측 상단을 기준으로 좌표값을 구함
